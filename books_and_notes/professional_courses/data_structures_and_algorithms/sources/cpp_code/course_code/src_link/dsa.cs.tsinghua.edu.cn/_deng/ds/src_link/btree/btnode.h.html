0009 #include "vector/vector.h"
0010 #define BTNodePosi(T) BTNode<T>* //B-树节点位置
0011 
0012 template <typename T> struct BTNode { //B-树节点模板类
0013 // 成员（为简化描述起见统一开放，读者可根据需要进一步封装）
0014    BTNodePosi(T) parent; //父节点
0015    Vector<T> key; //关键码向量
0016    Vector<BTNodePosi(T)> child; //孩子向量（其长度总比key多一）
0017 // 构造函数（注意：BTNode只能作为根节点创建，而且初始时有0个关键码和1个空孩子指针）
0018    BTNode() { parent = NULL; child.insert ( 0, NULL ); }
0019    BTNode ( T e, BTNodePosi(T) lc = NULL, BTNodePosi(T) rc = NULL ) {
0020       parent = NULL; //作为根节点，而且初始时
0021       key.insert ( 0, e ); //只有一个关键码，以及
0022       child.insert ( 0, lc ); child.insert ( 1, rc ); //两个孩子
0023       if ( lc ) lc->parent = this; if ( rc ) rc->parent = this;
0024    }
0025 };
