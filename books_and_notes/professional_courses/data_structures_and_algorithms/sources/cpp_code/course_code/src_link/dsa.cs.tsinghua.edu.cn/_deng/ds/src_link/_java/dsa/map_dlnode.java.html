0012 
0013 package  dsa;
0014 
0015 public class Map_DLNode implements Map {
0016    private List L;//存放条目的列表
0017    private EqualityTester T;//判等器
0018 
0019    //构造方法
0020    public Map_DLNode()
0021    {  this(new EqualityTesterDefault()); }
0022 
0023    //默认构造方法
0024    public Map_DLNode(EqualityTester t)
0025    {  L = new List_DLNode();  T = t; }
0026 
0027    /***************************** ADT方法 *****************************/
0028    //查询映射结构当前的规模
0029    public int getSize()
0030    { return L.getSize(); }
0031 
0032    //判断映射结构是否为空
0033    public boolean isEmpty()
0034    { return L.isEmpty(); }
0035 
0036    //若M中存在以key为关键码的条目，则返回该条目的数据对象；否则，返回null
0037    public Object get(Object key) {
0038       Iterator P = L.positions();
0039       while (P.hasNext()) {
0040          Position pos = (Position)P.getNext();
0041          Entry entry = (EntryDefault) pos.getElem();
0042          if (T.isEqualTo(entry.getKey(), key))  return entry.getValue();
0043       }
0044       return null;
0045    }
0046 
0047    //若M中不存在以key为关键码的条目，则将条目(key, value)加入到M中并返回null
0048    //否则，将已有条目的数据对象替换为value，并返回原先的数据对象
0049    public Object put(Object key, Object value)  {
0050       Iterator P = L.positions();
0051       while (P.hasNext()) {//逐一对比
0052          Position pos = (Position)P.getNext();//各个位置
0053          Entry entry = (EntryDefault) pos.getElem();//处的条目
0054          if (T.isEqualTo(entry.getKey(), key)) {//若发现key已出现在某个条目中，则
0055             Object oldValue = entry.getValue();//先保留该条目原先的数据对象
0056             L.replace(pos, new EntryDefault(key, value));//再替之以新数据对象
0057             return oldValue;//最后返回原先的数据对象。注意：返回null时的歧义
0058          }
0059       }//若此循环结束，说明key尚未在M中出现，因此
0060       L.insertFirst(new EntryDefault(key, value));//将新条目插至表首，并
0061       return null;//返回null标志
0062    }
0063 
0064    //若M中存在以key为关键码的条目，则删除之并返回其数据对象；否则，返回null
0065    public Object remove(Object key) {
0066       Iterator P = L.positions();
0067       while (P.hasNext()) {//逐一对比
0068          Position pos = (Position)P.getNext();//各个位置
0069          Entry entry = (EntryDefault) pos.getElem();//处的条目
0070          if (T.isEqualTo(entry.getKey(), key)) {//若发现key已出现在某个条目中，则
0071             Object oldValue = entry.getValue();//先保留该条目原先的数据对象
0072             L.remove(pos);//删除该条目
0073             return oldValue;//最后返回原先的数据对象。注意：返回null时的歧义
0074          }
0075       }//若此循环结束，说明key尚未在映射中出现，因此
0076       return null;//返回null标志
0077    }
0078 
0079    //返回M中所有条目的一个迭代器
0080    public Iterator entries()
0081    {  return new IteratorElement(L); }//直接利用List接口的方法生成元素迭代器
0082 }
