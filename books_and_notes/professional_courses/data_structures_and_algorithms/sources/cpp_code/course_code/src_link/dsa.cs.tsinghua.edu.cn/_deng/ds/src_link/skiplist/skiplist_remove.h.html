0009 template <typename K, typename V> bool Skiplist<K, V>::remove ( K k ) { //跳转表词条删除算法
0010    if ( empty() ) return false; //空表情况
0011    ListNode<Quadlist<Entry<K, V>>*>* qlist = first(); //从顶层Quadlist的
0012    QuadlistNode<Entry<K, V>>* p = qlist->data->first(); //首节点开始
0013    if ( !skipSearch ( qlist, p, k ) ) return false; //目标词条不存在，直接返回
0014    do { //若目标词条存在，则逐层拆除与之对应的塔
0015       QuadlistNode<Entry<K, V>>* lower = p->below; //记住下一层节点，并
0016       qlist->data->remove ( p ); //删除当前层节点，再
0017       p = lower; qlist = qlist->succ; //转入下一层
0018    } while ( qlist->succ ); //如上不断重复，直到塔基
0019    while ( !empty() && first()->data->empty() ) //逐一地
0020       List::remove ( first() ); //清除已可能不含词条的顶层Quadlist
0021    return true; //删除操作成功完成
0022 }
