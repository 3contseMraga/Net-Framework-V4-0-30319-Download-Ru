0011 bool collide ( int* solu, int k ) { //判断是否可以在当前行的第k列放置下一皇后
0012    nCheck++;
0013    for ( int i = 0; i < k; i++ ) {
0014       if ( solu[k] == solu[i] )       return true; //列
0015       if ( solu[k] - solu[i] == k - i ) return true; //对角线
0016       if ( solu[k] - solu[i] == i - k ) return true; //反对角线
0017    }
0018    return false;
0019 }
