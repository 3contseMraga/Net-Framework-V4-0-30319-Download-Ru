0009 /*
0010  * （有向）图的顶点结构接口
0011  * 无向图也可以看成是有向图，为此，只需将每条无向边替换为对称的一对有向边
0012  */
0013 
0014 package  dsa;
0015 
0016 public interface Vertex {
0017    //常量
0018    final static int UNDISCOVERED = 0;//尚未被发现的顶点
0019    final static int DISCOVERED = 1;//已被发现的顶点
0020    final static int VISITED = 2;//已访问过的顶点
0021 
0022    //返回当前顶点的信息
0023    public Object getInfo();
0024    //将当前顶点的信息更新为x，并返回原先的信息
0025    public Object setInfo(Object x);
0026 
0027    //返回当前顶点的出、入度
0028    public int outDeg();
0029    public int inDeg();
0030 
0031    //返回当前顶点所有关联边、关联边位置的迭代器
0032    public Iterator inEdges();
0033    public Iterator inEdgePositions();
0034    public Iterator outEdges();
0035    public Iterator outEdgePositions();
0036 
0037    //取当前顶点在所属的图的顶点集V中的位置
0038    public Position getVPosInV();
0039 
0040    //读取、设置顶点的状态（DFS + BFS）
0041    public int getStatus();
0042    public int setStatus(int s);
0043 
0044    //读取、设置顶点的时间标签（DFS）
0045    public int getDStamp();
0046    public int setDStamp(int s);
0047    public int getFStamp();
0048    public int setFStamp(int s);
0049 
0050    //读取、设置顶点至起点的最短距离（BFS或BestFS）
0051    public int getDistance();
0052    public int setDistance(int s);
0053 
0054    //读取、设置顶点在的DFS、BFS、BestFS或MST树中的父亲
0055    public Vertex getBFSParent();
0056    public Vertex setBFSParent(Vertex s);
0057 }
