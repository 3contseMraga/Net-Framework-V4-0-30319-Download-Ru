0010  * 基于有序（非升）列表实现的优先队列
0011  */
0012 
0013 package dsa;
0014 
0015 public class PQueue_SortedList implements PQueue {
0016    private List L;
0017    private Comparator C;
0018 
0019    //构造方法（使用默认比较器）
0020    public PQueue_SortedList()
0021    { this(new ComparatorDefault(), null); }
0022 
0023    //构造方法（使用指定比较器）
0024    public PQueue_SortedList(Comparator c)
0025    { this(c, null); }
0026 
0027    //构造方法（使用指定初始元素）
0028    public PQueue_SortedList(Sequence s)
0029    { this(new ComparatorDefault(), s); }
0030 
0031    //构造方法（使用指定比较器和初始元素）
0032    public PQueue_SortedList(Comparator c, Sequence s) {
0033       L = new List_DLNode();
0034       C = c;
0035       if (null != s)
0036          while (!s.isEmpty()) {
0037             Entry e = (Entry)s.removeFirst();
0038             insert(e.getKey(), e.getValue());
0039          }
0040    }
0041 
0042    //统计优先队列的规模
0043    public int getSize()
0044    { return L.getSize(); }
0045 
0046    //判断优先队列是否为空
0047    public boolean isEmpty()
0048    { return L.isEmpty(); }
0049 
0050    //若Q非空，则返回其中的最小条目（并不删除）;否则，报错
0051    public Entry getMin() throws ExceptionPQueueEmpty {
0052       if (L.isEmpty())
0053          throw new ExceptionPQueueEmpty("意外：优先队列空");
0054       return (Entry) L.last();
0055    }
0056 
0057    //将对象obj与关键码k合成一个条目，将其插入Q中，并返回该条目
0058    public Entry insert(Object key, Object obj) throws ExceptionKeyInvalid {
0059       Entry entry = new EntryDefault(key, obj);//创建一个新条目
0060       if (L.isEmpty()//若优先队列为空
0061       || (0 > C.compare(((Entry) (L.first().getElem())).getKey(), entry.getKey())))
0062          //或新条目是当前最大者
0063          L.insertFirst(entry);//则直接插入至表头
0064       else {//否则
0065          Position curPos = L.last();//从尾条目开始
0066          while (0 > C.compare(((Entry)(curPos.getElem())).getKey(), entry.getKey()))
0067             curPos = L.getPrev(curPos);//不断前移，直到第一个不小于entry的条目
0068          L.insertAfter(curPos, entry);//紧接该条目之后插入entry
0069       }
0070       return(entry);
0071    }
0072 
0073    //若Q非空，则从其中摘除关键码最小的条目，并返回该条目；否则，报错
0074    public Entry delMin() throws ExceptionPQueueEmpty {
0075       if (L.isEmpty())
0076          throw new ExceptionPQueueEmpty("意外：优先队列空");
0077       return (Entry) L.remove(L.last());
0078    }
0079 }
