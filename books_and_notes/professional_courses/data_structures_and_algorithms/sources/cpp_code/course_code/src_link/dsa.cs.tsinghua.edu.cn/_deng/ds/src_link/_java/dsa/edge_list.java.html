0009 /*
0010  * 基于邻接边表实现图的边结构
0011  */
0012 
0013 package  dsa;
0014 
0015 public class Edge_List implements Edge {
0016    //变量
0017    protected   Object info;//当前边中存放的数据元素
0018    protected Position ePosInE;//当前边在所属的图的边表中的位置
0019    protected Position vPosInV[];//当前边两个端点在顶点表中的位置
0020    protected Position ePosInI[];//当前边在其两个端点的关联边表中的位置
0021    //约定：第0（1）个顶点分别为尾（头）顶点
0022    //禁止头、尾顶点相同的边
0023    protected int type;//（经过遍历之后）边被归入的类别
0024 
0025    //构造方法：在图G中，生成一条从顶点u到v的新边（假定该边尚不存在）
0026    public Edge_List(Graph G, Vertex_List u, Vertex_List v, Object x) {
0027       info = x;//数据元素
0028       ePosInE = G.insert(this);//当前边在所属的图的边表中的位置
0029       vPosInV = new DLNode[2];//当前边两个端点在顶点表中的位置
0030       vPosInV[0] = u.getVPosInV(); vPosInV[1] = v.getVPosInV();
0031       ePosInI = new DLNode[2];//当前边在其两个端点的关联边表中的位置
0032       ePosInI[0] = u.outEdges.insertLast(this);//当前边加入u的邻接（出）边表
0033       ePosInI[1] = v.inEdges.insertLast(this);//当前边加入v的邻接（入）边表
0034       type = UNKNOWN;
0035    }
0036 
0037    //返回当前边的信息
0038    public Object getInfo() { return info; }
0039    //将当前边的信息更新为x，并返回原先的信息
0040    public Object setInfo(Object x) { Object e = info; info = x; return e; }
0041 
0042    //取当前边在所属的图的边集E中的位置
0043    public Position getEPosInE() { return ePosInE; }
0044    //取v[i]在顶点集V中的位置（i=0或1，分别对应于起点、终点）
0045    public Position getVPosInV(int i)   { return vPosInV[i]; }
0046    //当前边在其两个端点的关联边集I(v[i])中的位置
0047    public Position getEPosInI(int i) { return ePosInI[i]; }
0048 
0049    //读取、设置边的类别（针对遍历）
0050    public int getType() { return type; }
0051    public int setType(int t) { int tt = type; type = t; return tt; }
0052 
0053 }
