0012 
0013 package dsa;
0014 
0015 public class BinTree_LinkedList implements BinTree {
0016    protected BinTreePosition root;//根节点
0017 
0018    /**************************** 构造函数 ****************************/
0019    public BinTree_LinkedList()
0020    { this(null); }
0021 
0022    public BinTree_LinkedList(BinTreePosition r)
0023    { root = r; }
0024 
0025    /**************************** BinaryTree接口方法 ****************************/
0026    //返回树根
0027    public BinTreePosition getRoot()
0028    { return root; }
0029 
0030    //判断是否树空
0031    public boolean isEmpty()
0032    { return null == root; }
0033 
0034    //返回树的规模（即树根的后代数目）
0035    public int getSize()
0036    { return isEmpty() ? 0 : root.getSize(); }
0037 
0038    //返回树（根）的高度
0039    public int getHeight()
0040    {return isEmpty() ? -1 : root.getHeight(); }
0041 
0042    //前序遍历
0043    public Iterator elementsPreorder()
0044    { return root.elementsPreorder(); }
0045 
0046    //中序遍历
0047    public Iterator elementsInorder()
0048    { return root.elementsInorder(); }
0049 
0050    //后序遍历
0051    public Iterator elementsPostorder()
0052    { return root.elementsPostorder(); }
0053 
0054    //层次遍历
0055    public Iterator elementsLevelorder()
0056    { return root.elementsLevelorder(); }
0057 }
