0014 int testID = 0; //测试编号

0016 /******************************************************************************************
0017 * 测试栈
0018 ******************************************************************************************/
0019 template <typename T> //元素类型
0020 void testQueue ( int n ) {
0021    Queue<T> Q;
0022    printf ( "\n  ==== Test %2d. Growing queue\n", testID++ );
0023    while ( Q.size() < n ) {
0024       ( Q.empty() || ( 30 < dice ( 100 ) ) ) ?
0025          Q.enqueue ( (T) dice ( 2 * n ) ) : //70%入队
0026          Q.dequeue(); //30%出队
0027       print ( Q );
0028    }
0029    printf ( "\n  ==== Test %2d. Shrinking queue\n", testID++ );
0030    while ( !Q.empty() ) {
0031       ( 70 < dice ( 100 ) ) ?
0032          Q.enqueue ( (T) dice ( 2 * n ) ) : //30%入队
0033          Q.dequeue(); //70%出队
0034       print ( Q );
0035    }
0036 }
