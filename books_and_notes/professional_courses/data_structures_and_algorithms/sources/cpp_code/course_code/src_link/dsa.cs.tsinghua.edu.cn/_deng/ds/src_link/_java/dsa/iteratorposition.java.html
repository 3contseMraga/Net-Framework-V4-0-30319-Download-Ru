0009 /*
0010  * 基于列表实现的位置迭代器
0011  */
0012 
0013 package dsa;
0014 
0015 public class IteratorPosition implements Iterator {
0016    private List list;//列表
0017    private Position nextPosition;//当前（下一个）位置
0018 
0019    //默认构造方法
0020    public IteratorPosition() { list = null; }
0021 
0022    //构造方法
0023    public IteratorPosition(List L) {
0024       list = L;
0025       if (list.isEmpty())//若列表为空，则
0026          nextPosition = null;//当前位置置空
0027       else//否则
0028          nextPosition = list.first();//从第一个位置开始
0029    }
0030 
0031    //检查迭代器中是否还有剩余的位置
0032    public boolean hasNext() { return (nextPosition != null); }
0033 
0034    //返回迭代器中的下一位置
0035    public Object getNext() throws ExceptionNoSuchElement {
0036       if (!hasNext()) throw new ExceptionNoSuchElement("意外：没有下一位置");
0037       Position currentPosition = nextPosition;
0038       if (currentPosition == list.last())//若已到达尾位置，则
0039          nextPosition = null;//不再有下一个位置
0040       else//否则
0041          nextPosition = list.getNext(currentPosition);//转向下一位置
0042       return currentPosition;
0043    }
0044 }
