0018 /******************************************************************************************
0019  * 基础BinTree
0020  ******************************************************************************************/
0021 template <typename T> //元素类型
0022 void UniPrint::p ( BinTree<T> & bt ) { //引用
0023    printf ( "%s[%d]*%d:\n", typeid ( bt ).name(), &bt, bt.size() ); //基本信息
0024    Bitmap* branchType = new Bitmap; //记录当前节点祖先的方向

0026    release ( branchType ); printf ( "\n" );
0027 }

0076 /******************************************************************************************
0077  * 二叉树各种派生类的统一打印
0078  ******************************************************************************************/
0079 template <typename T> //元素类型
0080 static void printBinTree ( BinNodePosi(T) bt, int depth, int type, Bitmap* bType ) {
0081    if ( !bt ) return;
0082    if ( -1 < depth ) //设置当前层的拐向标志
0083       R_CHILD == type ? bType->set ( depth ) : bType->clear ( depth );
0084    printBinTree ( bt->rc, depth + 1, R_CHILD, bType ); //右子树（在上）
0085    print ( bt ); printf ( " *" );
0086    for ( int i = -1; i < depth; i++ ) //根据相邻各层
0087       if ( ( 0 > i ) || bType->test ( i ) == bType->test ( i + 1 ) ) //的拐向是否一致，即可确定
0088          printf ( "      " ); //是否应该
0089       else  printf ( "│    " ); //打印横线
0090    switch ( type ) {
0091       case  R_CHILD  :  printf ( "┌─" );  break;
0092       case  L_CHILD  :  printf ( "└─" );  break;
0093       default        :  printf ( "──" );  break; //root
0094    }
0095    print ( bt );
0096 #if defined(DSA_HUFFMAN)
0097    if ( IsLeaf ( *bt ) ) bType->print ( depth + 1 ); //输出Huffman编码
0098 #endif
0099    printf ( "\n" );
0100    printBinTree ( bt->lc, depth + 1, L_CHILD, bType ); //左子树（在下）
0101 }
