0010  * （有向）图基于BFS的最短距离算法
0011  */
0012 
0013 package dsa;
0014 
0015 public class BFSDistance extends BFS {
0016    //构造方法
0017    public BFSDistance(Graph g) { super(g); }
0018 
0019    //顶点访问操作：在本算法中，info是顶点v的前驱
0020    protected Object visit(Vertex v, Object info)   {
0021       if (null == info)//v为BFS的起始顶点
0022          v.setDistance(0);
0023       else
0024          v.setDistance(((Vertex)info).getDistance() + 1); //设置v到s的距离 = 前驱的距离+1
0025       return null;
0026    }
0027 
0028    //基于BFS实现的最短距离算法：s为起始顶点，info向算法传递参数
0029    public Object algorithm(Vertex s, Object info) {
0030       reset(s);
0031       traverse(s, info);//BFS：到起点的最短距离记录在各顶点的distance域中
0032       return null;
0033    }
0034 }
