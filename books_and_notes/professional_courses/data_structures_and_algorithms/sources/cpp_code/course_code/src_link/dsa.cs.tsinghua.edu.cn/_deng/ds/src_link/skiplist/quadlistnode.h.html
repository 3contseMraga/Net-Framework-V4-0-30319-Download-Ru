0009 #include "Entry/Entry.h"
0010 #define QlistNodePosi(T)  QuadlistNode<T>* //跳转表节点位置
0011 
0012 template <typename T> struct QuadlistNode { //QuadlistNode模板类
0013    T entry; //所存词条
0014    QlistNodePosi(T) pred;  QlistNodePosi(T) succ; //前驱、后继
0015    QlistNodePosi(T) above; QlistNodePosi(T) below; //上邻、下邻
0016    QuadlistNode //构造器
0017    ( T e = T(), QlistNodePosi(T) p = NULL, QlistNodePosi(T) s = NULL,
0018      QlistNodePosi(T) a = NULL, QlistNodePosi(T) b = NULL )
0019       : entry ( e ), pred ( p ), succ ( s ), above ( a ), below ( b ) {}
0020    QlistNodePosi(T) insertAsSuccAbove //插入新节点，以当前节点为前驱，以节点b为下邻
0021    ( T const& e, QlistNodePosi(T) b = NULL );
0022 };
