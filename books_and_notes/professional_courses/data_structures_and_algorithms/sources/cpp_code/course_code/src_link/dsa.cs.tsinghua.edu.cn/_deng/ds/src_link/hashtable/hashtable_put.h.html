0009 template <typename K, typename V> bool Hashtable<K, V>::put ( K k, V v ) { //散列表词条插入
0010    if ( ht[probe4Hit ( k ) ] ) return false; //雷同元素不必重复插入
0011    int r = probe4Free ( k ); //为新词条找个空桶（只要装填因子控制得当，必然成功）
0012    ht[r] = new Entry<K, V> ( k, v ); ++N; //插入（注意：懒惰删除标记无需复位）
0013    if ( N * 2 > M ) rehash(); //装填因子高于50%后重散列
0014    return true;
0015 }
