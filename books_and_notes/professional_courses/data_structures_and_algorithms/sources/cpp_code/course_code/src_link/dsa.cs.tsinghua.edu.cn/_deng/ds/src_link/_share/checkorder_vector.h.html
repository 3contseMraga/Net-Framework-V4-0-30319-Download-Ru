0011 template <typename T> void checkOrder ( Vector<T> & V ) { //判断向量是否整体有序
0012    int unsorted = 0; //逆序计数器
0013    V.traverse ( CheckOrder<T> ( unsorted, V[0] ) ); //进行遍历
0014    if ( 0 < unsorted )
0015       printf ( "Unsorted with %d adjacent inversion(s)\n", unsorted );
0016    else
0017       printf ( "Sorted\n" );
0018 }
