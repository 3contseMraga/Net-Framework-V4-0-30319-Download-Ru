0009 __int64 gcdEU_R ( __int64 a, __int64 b ) { //assert: 0 <= min(a, b)
0010    return ( 0 == b ) ? a : gcdEU_R ( b, a % b );
0011 }

0013 __int64 gcdEU ( __int64 a, __int64 b ) { //assert: 0 <= min(a, b)
0014    while ( true ) {
0015       if ( 0 == a ) return b;
0016       if ( 0 == b ) return a;
0017       if ( a > b ) a %= b;
0018       else       b %= a;
0019    }
0020 }
