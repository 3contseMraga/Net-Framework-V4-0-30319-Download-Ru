0009 /*
0010  * 二叉树节点ADT接口
0011  */
0012 
0013 package dsa;
0014 
0015 public interface BinTreePosition extends Position {
0016    //判断是否有父亲（为使代码描述简洁）
0017    public boolean hasParent();
0018    //返回当前节点的父节点
0019    public BinTreePosition getParent();
0020    //设置当前节点的父节点
0021    public void setParent(BinTreePosition p);
0022 
0023    //判断是否为叶子
0024    public boolean isLeaf();
0025 
0026    //判断是否为左孩子（为使代码描述简洁）
0027    public boolean isLChild();
0028 
0029    //判断是否有左孩子（为使代码描述简洁）
0030    public boolean hasLChild();
0031    //返回当前节点的左孩子
0032    public BinTreePosition getLChild();
0033    //设置当前节点的左孩子（注意：this.lChild和c.parent都不一定为空）
0034    public void setLChild(BinTreePosition c);
0035 
0036    //判断是否为右孩子（为使代码描述简洁）
0037    public boolean isRChild();
0038    //判断是否有右孩子（为使代码描述简洁）
0039    public boolean hasRChild();
0040    //返回当前节点的右孩子
0041    public BinTreePosition getRChild();
0042    //设置当前节点的右孩子（注意：this.rChild和c.parent都不一定为空）
0043    public void setRChild(BinTreePosition c);
0044 
0045    //返回当前节点后代元素的数目
0046    public int getSize();
0047    //在孩子发生变化后，更新当前节点及其祖先的规模
0048    public void updateSize();
0049 
0050    //返回当前节点的高度
0051    public int getHeight();
0052    //在孩子发生变化后，更新当前节点及其祖先的高度
0053    public void updateHeight();
0054 
0055    //返回当前节点的深度
0056    public int getDepth();
0057    //在父亲发生变化后，更新当前节点及其后代的深度
0058    public void updateDepth();
0059 
0060    //按照中序遍历的次序，找到当前节点的直接前驱
0061    public BinTreePosition getPrev();
0062 
0063    //按照中序遍历的次序，找到当前节点的直接后继
0064    public BinTreePosition getSucc();
0065 
0066    //断绝当前节点与其父亲的父子关系
0067    //返回当前节点
0068    public BinTreePosition secede();
0069 
0070    //将节点c作为当前节点的左孩子
0071    public BinTreePosition attachL(BinTreePosition c);
0072 
0073    //将节点c作为当前节点的右孩子
0074    public BinTreePosition attachR(BinTreePosition c);
0075 
0076    //前序遍历
0077    public Iterator elementsPreorder();
0078 
0079    //中序遍历
0080    public Iterator elementsInorder();
0081 
0082    //后序遍历
0083    public Iterator elementsPostorder();
0084 
0085    //层次遍历
0086    public Iterator elementsLevelorder();
0087 }
