0010  * （有向）带权图的单源点最短路径算法
0011  */
0012 
0013 package dsa;
0014 
0015 public class BestFSDijkstra extends BestFS {
0016    //构造方法
0017    public BestFSDijkstra(Graph g) { super(g); }
0018 
0019    //更新尚未访问的顶点到源点的最短距离
0020    protected void updateDistanceAfter(Vertex v) {
0021       for (Iterator it = v.outEdges(); it.hasNext();) {//检查与顶点v
0022          Edge e = (Edge)it.getNext();//通过边e = (v, w)
0023          Vertex w = (Vertex)e.getVPosInV(1).getElem();//相联的每一顶点w
0024          int   weight = ((Integer)e.getInfo()).intValue();//根据边(v, w)的权重
0025          if (w.getDistance() > v.getDistance() + weight) {//取原距离与新距离中的小者
0026             w.setDistance(v.getDistance() + weight);
0027             w.setBFSParent(v);
0028          }
0029       }
0030    }
0031 }
