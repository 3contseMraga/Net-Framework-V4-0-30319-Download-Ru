0012 int testBitmap ( int n ) {
0013    bool* B = new bool[n]; memset ( B, 0, n * sizeof ( bool ) ); //常规位图，创建后随即O(n)时间初始化
0014    Bitmap M ( n );
0015    for ( int i = 0; i < 9 * n; i++ ) {
0016       Rank k = dice ( n );
0017       printf ( "set(%d) ...", k ); //set(k)
0018       B[k] = true; M.set ( k );
0019       printf ( "done\n CRC: " );
0020       for ( int j = 0; j < n; j++ )
0021          printf ( "%6c", B[j] == M.test ( j ) ? ' ' : '!' );
0022       printf ( "\n B[]: " );
0023       for ( int j = 0; j < n; j++ )
0024          printf ( "%6c", B[j] ? 'x' : '.' );
0025       printf ( "\n M[]: " );
0026       for ( int j = 0; j < n; j++ )
0027          printf ( "%6c", M.test ( j ) ? 'x' : '.' );
0028       printf ( "\n\n\n" );
0029    }
0030    delete [] B;
0031    return 0;
0032 }
