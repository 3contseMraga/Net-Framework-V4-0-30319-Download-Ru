0009 template <typename T> //轴点构造算法：通过调整元素位置构造区间[lo, hi)的轴点，并返回其秩
0010 Rank Vector<T>::partition ( Rank lo, Rank hi ) { //版本B1：版本B的等价形式，可直接转至与版本A等价的版本A1
0011    swap ( _elem[lo], _elem[ lo + rand() % ( hi - lo ) ] ); //任选一个元素与首元素交换
0012    hi--; T pivot = _elem[lo]; //以首元素为候选轴点――经以上交换，等效于随机选取
0013    while ( lo < hi ) { //从向量的两端交替地向中间扫描
0014       while ( ( lo < hi ) && ( pivot < _elem[hi] ) ) //在大于pivot的前提下
0015          hi--; //向左拓展右端子向量
0016       if ( lo < hi ) _elem[lo++] = _elem[hi]; //不大于pivot者归入左端子向量
0017       while ( ( lo < hi ) && ( _elem[lo] < pivot ) ) //在小于pivot的前提下
0018          lo++; //向右拓展左端子向量
0019       if ( lo < hi ) _elem[hi--] = _elem[lo]; //不小于pivot者归入右端子向量
0020    } //assert: lo == hi
0021    _elem[lo] = pivot; //将备份的轴点记录置于前、后子向量之间
0022    return lo; //返回轴点的秩
0023 }
