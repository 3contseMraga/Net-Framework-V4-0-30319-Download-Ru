0009 template <typename T> void List<T>::reverse() { //前后倒置
0010    if ( _size < 2 ) return; //平凡情况
0011    ListNodePosi(T) p; ListNodePosi(T) q;
0012    for ( p = header, q = p->succ; p != trailer; p = q, q = p->succ )
0013       p->pred = q; //自前向后，依次颠倒各节点的前驱指针
0014    trailer->pred = NULL; //单独设置尾节点的前驱指针
0015    for ( p = header, q = p->pred; p != trailer; p = q, q = p->pred )
0016       q->succ = p; //自前向后，依次颠倒各节点的后继指针
0017    header->succ = NULL; //单独设置头节点的后继指针
0018    swap ( header, trailer ); //头、尾节点互换
0019 }
