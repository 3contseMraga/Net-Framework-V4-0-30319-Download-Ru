0009 template <typename T> //根据相对优先级确定适宜的方式，合并以a和b为根节点的两个左式堆
0010 static BinNodePosi(T) merge ( BinNodePosi(T) a, BinNodePosi(T) b ) {
0011    if ( ! a ) return b; //退化情况
0012    if ( ! b ) return a; //退化情况
0013    if ( lt ( a->data, b->data ) ) swap ( a, b ); //一般情况：首先确保b不大
0014    ( a->rc = merge ( a->rc, b ) )->parent = a; //将a的右子堆，与b合并
0015    if ( !a->lc || a->lc->npl < a->rc->npl ) //若有必要
0016       swap ( a->lc, a->rc ); //交换a的左、右子堆，以确保右子堆的npl不大
0017    a->npl = a->rc ? a->rc->npl + 1 : 1; //更新a的npl
0018    return a; //返回合并后的堆顶
0019 } //本算法只实现结构上的合并，堆的规模须由上层调用者负责更新
