0009 template <typename T> void quickSelect ( Vector<T> & A, Rank k ) { //基于快速划分的k选取算法
0010    for ( Rank lo = 0, hi = A.size() - 1; lo < hi; ) {
0011       Rank i = lo, j = hi; T pivot = A[lo];
0012       while ( i < j ) { //O(hi - lo + 1) = O(n)
0013          while ( ( i < j ) && ( pivot <= A[j] ) ) j--; A[i] = A[j];
0014          while ( ( i < j ) && ( A[i] <= pivot ) ) i++; A[j] = A[i];
0015       } //assert: i == j
0016       A[i] = pivot;
0017       if ( k <= i ) hi = i - 1;
0018       if ( i <= k ) lo = i + 1;
0019    } //A[k] is now a pivot
0020 }
