0012 /******************************************************************************************
0013  * 检查表达式括号是否匹配
0014  ******************************************************************************************/
0015 int main ( int argc, char* argv[] ) {
0016 // 检查参数
0017    if ( 2 > argc ) { //至少要提供一个待检查的字符串
0018       fprintf ( stderr, "\nUsage: %s <string#1> <string#2> ... \n", argv[0] );
0019       fprintf ( stderr, "For example %s (a[i-1][j+1])+a[i+1][j-1])*2\n", argv[0] );
0020       return -1;
0021    }
0022 // 逐一检查各串是否匹配
0023    for ( int i = 1; i < argc; i++ ) {
0024       char* s = argv[i];   printf ( "\n检查：%s\n", s );
0025       paren ( s, 0, strlen ( s ) - 1 ) ?
0026       printf ( "\n%s: 匹配\a\n", s ) :
0027       printf ( "\n%s: 不匹配\a\a\n", s );
0028       getchar();
0029    }
0030    return 0;
0031 }
