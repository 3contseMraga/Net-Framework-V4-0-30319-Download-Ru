0009 template <typename T> //将e紧随当前节点之后插入于当前节点所属列表（设有哨兵尾节点trailer）
0010 ListNodePosi(T) ListNode<T>::insertAsSucc ( T const& e ) {
0011    ListNodePosi(T) x = new ListNode ( e, this, succ ); //创建新节点
0012    succ->pred = x; succ = x; //设置逆向链接
0013    return x; //返回新节点的位置
0014 }
