0010  * 基于单链表实现队列结构

0012 
0013 package dsa;
0014 
0015 public class Queue_List implements Queue {
0016    protected Node head;//指向表首元素
0017    protected Node tail;//指向表末元素
0018    protected int size;//队列中元素的数目
0019 
0020    //构造方法（空队列）
0021    public Queue_List()
0022    {  head = tail = null;  size = 0; }
0023 
0024    //查询当前队列的规模
0025    public int getSize()
0026    {  return size; }
0027 
0028    //判断队列是否为空
0029    public boolean isEmpty()
0030    {  return (0 == size) ? true : false; }
0031 
0032    //入队
0033    public void enqueue(Object obj) {
0034       Node node = new Node();
0035       node.setElem(obj);
0036       node.setNext(null);//新节点作为末节点插入
0037       if (0 == size) head = node;//若此前队列为空，则直接插入
0038       else                 tail.setNext(node);//否则，将新节点接至队列末端
0039       tail = node;//更新指向末节点引用
0040       size++;//更新规模
0041    }
0042 
0043    //出队
0044    public Object dequeue() throws ExceptionQueueEmpty {
0045       if (0 == size)
0046          throw new ExceptionQueueEmpty("意外：队列空");
0047       Object obj = head.getElem();
0048       head = head.getNext();
0049       size--;
0050       if (0 == size) tail = null;//若队列已空，须将末节点引用置空
0051       return obj;
0052    }
0053 
0054    //取（并不删除）队首元素
0055    public Object front() throws ExceptionQueueEmpty {
0056       if (isEmpty())
0057          throw new ExceptionQueueEmpty("意外：队列空");
0058       return head.getElem();
0059    }
0060 
0061    //遍历（不属于ADT）
0062    public void Traversal() {
0063       Node  p = head;
0064       while (null != p) {
0065          System.out.print(p.getElem() + " ");
0066          p = p.getNext();
0067       }
0068    }
0069 }
