0009 __int64 gcdCN ( __int64 a, __int64 b ) { //assert: 0 < min(a, b)
0010    int r = 0; //a和b的2^r形式的公因子
0011    while ( ! ( ( a & 1 ) || ( b & 1 ) ) ) { //若a和b都是偶数
0012       a >>= 1; b >>= 1; r ++; //则同时除2（右移），并累加至r
0013    } //以下，a和b至多其一为偶
0014    while ( 1 ) {
0015       while ( ! ( a & 1 ) ) a >>= 1; //若a偶（b奇），则剔除a的所有因子2
0016       while ( ! ( b & 1 ) ) b >>= 1; //若b偶（a奇），则剔除b的所有因子2
0017       ( a > b ) ? a = a - b : b = b - a; //简化为：gcd(max(a, b) - min(a, b), min(a, b))
0018       if ( 0 == a ) return b << r; //简化至平凡情况：gcd(0, b) = b
0019       if ( 0 == b ) return a << r; //简化至平凡情况：gcd(a, 0) = a
0020    }
0021 }
