0013 package dsa;
0014 
0015 public interface Tree {
0016    //返回当前节点中存放的对象
0017    public Object getElem();
0018    //将对象obj存入当前节点，并返回此前的内容
0019    public Object setElem(Object obj);
0020 
0021    //返回当前节点的父节点
0022    public TreeLinkedList getParent();
0023 
0024    //返回当前节点的长子
0025    public TreeLinkedList getFirstChild();
0026 
0027    //返回当前节点的最大弟弟
0028    public TreeLinkedList getNextSibling();
0029 
0030    //返回当前节点后代元素的数目，即以当前节点为根的子树的规模
0031    public int getSize();
0032 
0033    //返回当前节点的高度
0034    public int getHeight();
0035 
0036    //返回当前节点的深度
0037    public int getDepth();
0038 }
