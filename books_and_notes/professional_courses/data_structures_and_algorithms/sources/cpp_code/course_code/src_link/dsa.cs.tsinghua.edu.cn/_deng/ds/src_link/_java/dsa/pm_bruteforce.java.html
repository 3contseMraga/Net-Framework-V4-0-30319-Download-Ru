0009 /*
0010  * 串模式匹配：蛮力算法
0011  * 若返回位置i > length(T) - length(P)，则说明失配
0012  * 否则，i为匹配位置
0013  */
0014 import dsa.*;
0015 import java.io.*;
0016 
0017 public class PM_BruteForce {
0018    //////////////////////////////////////////////////////////////////////////
0019    // T: 0     1     .     .     .     i     i+1   .     .     .     i+j   .     .     n-1
0020    //    --------------------|-------------------|------------
0021    // P:                               0     1     .     .     .     j     .     .
0022    //                                  |-------------------|
0023    //////////////////////////////////////////////////////////////////////////
0024    public static int PM(String T, String P) {
0025       int      i;//模式串相对于文本串的起始位置
0026       int      j;//模式串当前字符的地址
0027       for (i = 0; i <= T.length() - P.length(); i++) { //文本串从第i个字符起，与
0028          for (j = 0; j < P.length(); j++) { //模式串的当前字符逐次比较
0029             if (T.charAt(i + j) != P.charAt(j))   break; //若失配，模式串右移一个字符
0030          }
0031          if (j >= P.length()) break;//找到匹配子串
0032       }
0033       return(i);
0034    }
0035 }
