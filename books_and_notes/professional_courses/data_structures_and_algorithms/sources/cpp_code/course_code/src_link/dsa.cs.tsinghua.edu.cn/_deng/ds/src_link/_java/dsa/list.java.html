0013 package dsa;
0014 
0015 public interface List {
0016    //查询列表当前的规模
0017    public int getSize();
0018 
0019    //判断列表是否为空
0020    public boolean isEmpty();
0021 
0022    //返回第一个元素（的位置）
0023    public Position first();
0024 
0025    //返回最后一个元素（的位置）
0026    public Position last();
0027 
0028    //返回紧接给定位置之后的元素（的位置）
0029    public Position getNext(Position p)
0030    throws ExceptionPositionInvalid, ExceptionBoundaryViolation;
0031 
0032    //返回紧靠给定位置之前的元素（的位置）
0033    public Position getPrev(Position p)
0034    throws ExceptionPositionInvalid, ExceptionBoundaryViolation;
0035 
0036    //将e作为第一个元素插入列表
0037    public Position insertFirst(Object e);
0038 
0039    //将e作为最后一个元素插入列表
0040    public Position insertLast(Object e);
0041 
0042    //将e插入至紧接给定位置之后的位置
0043    public Position insertAfter(Position p, Object e)
0044    throws ExceptionPositionInvalid;
0045 
0046    //将e插入至紧靠给定位置之前的位置
0047    public Position insertBefore(Position p, Object e)
0048    throws ExceptionPositionInvalid;
0049 
0050    //删除给定位置处的元素，并返回之
0051    public Object remove(Position p)
0052    throws ExceptionPositionInvalid;
0053 
0054    //删除首元素，并返回之
0055    public Object removeFirst();
0056 
0057    //删除末元素，并返回之
0058    public Object removeLast();
0059 
0060    //将处于给定位置的元素替换为新元素，并返回被替换的元素
0061    public Object replace(Position p, Object e)
0062    throws ExceptionPositionInvalid;
0063 
0064    //位置迭代器
0065    public Iterator positions();
0066 
0067    //元素迭代器
0068    public Iterator elements();
0069 }
