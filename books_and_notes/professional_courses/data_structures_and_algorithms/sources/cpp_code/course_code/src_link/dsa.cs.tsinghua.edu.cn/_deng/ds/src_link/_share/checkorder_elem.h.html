0009 template <typename T> struct CheckOrder { //函数对象：判断一个T类对象是否局部有序
0010    T pred; int& u;
0011    CheckOrder ( int& unsorted, T& first ) : pred ( first ), u ( unsorted ) { }
0012    virtual void operator() ( T& e ) { if ( pred > e ) u++; pred = e; }
0013 };
