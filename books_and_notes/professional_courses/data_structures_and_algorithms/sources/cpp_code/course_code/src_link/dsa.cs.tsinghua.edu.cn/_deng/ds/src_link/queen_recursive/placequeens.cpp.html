0011 /******************************************************************************************
0012  * n皇后算法（递归版）
0013  * 初始启动：placeQueens(nQueen, 0);
0014  ******************************************************************************************/
0015 void placeQueens ( int nQueen, int k ) { //放置n个皇后中的第k个（假设0至k-1均已就位）
0016    static int solu[QUEEN_MAX]; //解法
0017    if ( nQueen <= k ) { //若所有皇后都已就位，则
0018       nSolu++; displaySolution ( solu, nQueen ); //输出可行解
0019    } else //否则
0020       for ( int i = 0; i < nQueen; i++ ) { //依次
0021          solu[k] = i; //试着将当前皇后放在（当前行的）各列上
0022          if ( !collide ( solu, k ) ) //若没有冲突，则
0023             placeQueens ( nQueen, k + 1 ); //进而考虑下一皇后
0024       } //思考：这里的回溯，通过什么机制实现？
0025 }
