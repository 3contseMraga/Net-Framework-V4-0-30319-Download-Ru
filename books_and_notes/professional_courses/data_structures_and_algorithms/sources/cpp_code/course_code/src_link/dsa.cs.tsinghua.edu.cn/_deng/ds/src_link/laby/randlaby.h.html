0009 void randLaby() { //生成随机的迷宫
0010    labySize = LABY_MAX / 2 + rand() % ( LABY_MAX / 2 );
0011    for ( int i = 0; i < labySize; i++ )
0012       for ( int j = 0; j < labySize; j++ ) {
0013          laby[i][j].x = i;
0014          laby[i][j].y = j;
0015          laby[i][j].incoming   =
0016             laby[i][j].outgoing   =   UNKNOWN;
0017          laby[i][j].status = WALL; //边界格点必须是墙
0018       }
0019    for ( int i = 1; i < labySize - 1; i++ )
0020       for ( int j = 1; j < labySize - 1; j++ )
0021          if ( rand() % 4 ) laby[i][j].status = AVAILABLE; //75%的格点为空可用
0022    startCell = &laby[rand() % ( labySize-2 ) + 1][rand() % ( labySize-2 ) + 1];
0023    goalCell  = &laby[rand() % ( labySize-2 ) + 1][rand() % ( labySize-2 ) + 1];
0024    startCell->status = goalCell->status = AVAILABLE; //起始格点必须可用
0025 }
