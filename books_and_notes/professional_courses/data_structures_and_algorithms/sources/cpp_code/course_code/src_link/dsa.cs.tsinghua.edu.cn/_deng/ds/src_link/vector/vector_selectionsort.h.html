0009 template <typename T> //向量选择排序
0010 void Vector<T>::selectionSort ( Rank lo, Rank hi ) { //0 <= lo < hi <= size
0011    while ( lo < --hi )
0012       swap ( _elem[max ( lo, hi ) ], _elem[hi] ); //将[hi]与[lo, hi]中的最大者交换
0013 }
0014 
0015 template <typename T>
0016 Rank Vector<T>::max ( Rank lo, Rank hi ) { //在[lo, hi]内找出最大者
0017    Rank mx = hi;
0018    while ( lo < hi-- ) //逆向扫描
0019       if ( _elem[hi] > _elem[mx] ) //且严格比较
0020          mx = hi; //故能在max有多个时保证后者优先，进而保证selectionSort稳定
0021    return mx;
0022 }
