0010  * 单链表节点类
0011  */
0012 
0013 package dsa;
0014 
0015 public class Node implements Position {
0016    private Object element;//数据对象
0017    private Node next;//指向后继节点
0018 
0019    /**************************** 构造函数 ****************************/
0020    public Node()
0021    {  this(null, null); }//指向数据对象、后继节点的引用都置空
0022 
0023    public Node(Object e, Node n)
0024    {  element = e;   next = n;   }//指定数据对象及后继节点
0025 
0026    /**************************** Position接口方法 ****************************/
0027    //返回存放于该位置的元素
0028    public Object getElem() { return element; }
0029 
0030    //将给定元素存放至该位置，返回此前存放的元素
0031    public Object setElem(Object e)
0032    { Object oldElem = element;   element = e;   return oldElem; }
0033 
0034    /**************************** 单链表节点方法 ****************************/
0035    //取当前节点的后继节点
0036    public Node getNext()
0037    { return next; }
0038 
0039 
0040    //修改当前节点的后继节点
0041    public void setNext(Node newNext)
0042    { next = newNext; }
0043 }
