0009 /*
0010  * 基于位置接口实现的双向链表节点类
0011  */
0012 
0013 package dsa;
0014 
0015 public class DLNode implements Position {
0016    private Object element;//数据对象
0017    private DLNode prev;//指向前驱节点
0018    private DLNode next;//指向后继节点
0019 
0020    /**************************** 构造函数 ****************************/
0021    public DLNode()
0022    { this(null, null, null); }
0023 
0024    public DLNode(Object e, DLNode p, DLNode n)
0025    { element = e; prev = p; next = n; }
0026    //注意三个参数的次序：数据对象、前驱节点、后继节点
0027 
0028    /**************************** Position接口方法 ****************************/
0029    //返回存放于该位置的元素
0030    public Object getElem()
0031    { return element; }
0032 
0033    //将给定元素存放至该位置，返回此前存放的元素
0034    public Object setElem(Object e)
0035    { Object oldElem = element;   element = e;   return oldElem; }
0036 
0037    /**************************** 双向链表节点方法 ****************************/
0038    //找到后继位置
0039    public DLNode getNext()
0040    { return next; }
0041 
0042    //找到前驱位置
0043    public DLNode getPrev()
0044    { return prev; }
0045 
0046    //修改后继位置
0047    public void setNext(DLNode newNext)
0048    { next = newNext; }
0049 
0050    //修改前驱位置
0051    public void setPrev(DLNode newPrev)
0052    { prev = newPrev; }
0053 }
