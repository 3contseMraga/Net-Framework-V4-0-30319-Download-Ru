0009 /*
0010  * 基于链表实现的BST节点类
0011  */
0012 
0013 package dsa;
0014 
0015 public class BSTreeNode extends BinTreeNode implements BinTreePosition, Entry {
0016    /**************************** 构造方法 ****************************/
0017    public BSTreeNode()
0018    { super(); }
0019 
0020    public BSTreeNode(
0021       Object e,//节点内容
0022       BinTreePosition p,//父节点
0023       boolean asLChild,//是否作为父节点的左孩子
0024       BinTreePosition l,//左孩子
0025       BinTreePosition r)//右孩子
0026    {  super(e, p, asLChild, l, r); }
0027 
0028    /**************************** 实现Entry接口的方法 ****************************/
0029    //返回当前节点的关键码
0030    public Object getKey()
0031    { return ((Entry)getElem()).getKey(); }
0032 
0033    //修改条目的关键码，返回此前存放的关键码
0034    public Object setKey(Object k)
0035    { return ((Entry)getElem()).setKey(k); }
0036 
0037    //取条目的数据对象
0038    public Object getValue()
0039    { return ((Entry)getElem()).getValue(); }
0040 
0041    //修改条目的数据对象，返回此前存放的数据对象
0042    public Object setValue(Object v)
0043    { return ((Entry)getElem()).setValue(v); }
0044 }
