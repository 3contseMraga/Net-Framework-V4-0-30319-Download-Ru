0009 int* buildNext ( char* P ) { //构造模式串P的next表
0010    size_t m = strlen ( P ), j = 0; //“主”串指针
0011    int* N = new int[m]; //next表
0012    int t = N[0] = -1; //模式串指针
0013    while ( j < m - 1 )
0014       if ( 0 > t || P[j] == P[t] ) { //匹配
0015          j ++; t ++;
0016          N[j] = t; //此句可改进...
0017       } else //失配
0018          t = N[t];
0019    return N;
0020 }
