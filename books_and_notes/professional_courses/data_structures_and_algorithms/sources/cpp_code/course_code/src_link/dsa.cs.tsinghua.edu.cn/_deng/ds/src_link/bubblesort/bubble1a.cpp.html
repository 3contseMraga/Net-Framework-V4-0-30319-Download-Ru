0009 void bubblesort1A ( int A[], int n ) { //起泡排序算法（版本1A）：0 <= n
0010    bool sorted = false; //整体排序标志，首先假定尚未排序
0011    while ( !sorted ) { //在尚未确认已全局排序之前，逐趟进行扫描交换
0012       sorted = true; //假定已经排序
0013       for ( int i = 1; i < n; i++ ) { //自左向右逐对检查当前范围A[0, n)内的各相邻元素
0014          if ( A[i - 1] > A[i] ) { //一旦A[i - 1]与A[i]逆序，则
0015             swap ( A[i - 1], A[i] ); //交换之，并
0016             sorted = false; //因整体排序不能保证，需要清除排序标志
0017          }
0018       }
0019       n--; //至此末元素必然就位，故可以缩短待排序序列的有效长度
0020    }
0021 } //借助布尔型标志位sorted，可及时提前退出，而不致总是蛮力地做n - 1趟扫描交换
