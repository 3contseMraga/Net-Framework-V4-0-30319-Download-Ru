0010  * 基于单链表实现栈结构
0011  */
0012 
0013 package dsa;
0014 
0015 public class Stack_List implements Stack {
0016    protected Node top;//指向栈顶元素
0017    protected int size;//栈中元素的数目
0018 
0019    //构造方法（空栈）
0020    public Stack_List()
0021    {  top = null; size = 0; }
0022 
0023    //查询当前栈的规模
0024    public int getSize()
0025    {  return size; }
0026 
0027    //判断是否栈空
0028    public boolean isEmpty()
0029    {  return (top == null) ? true : false; }
0030 
0031    //压栈
0032    public void push(Object elem) {
0033       Node v = new Node(elem, top);//创建一个新节点，将其作为首节点插入
0034       top = v;//更新首节点引用
0035       size++;//更新规模记录
0036    }
0037 
0038    //读取（但不删除）栈顶
0039    public Object top() throws ExceptionStackEmpty {
0040       if (isEmpty())
0041          throw new ExceptionStackEmpty("意外：栈空");
0042       return top.getElem();
0043    }
0044 
0045    //弹出栈顶
0046    public Object pop() throws ExceptionStackEmpty {
0047       if (isEmpty())
0048          throw new ExceptionStackEmpty("意外：栈空");
0049       Object temp = top.getElem();
0050       top = top.getNext();//更新首节点引用
0051       size--;//更新规模记录
0052       return temp;
0053    }
0054 }
