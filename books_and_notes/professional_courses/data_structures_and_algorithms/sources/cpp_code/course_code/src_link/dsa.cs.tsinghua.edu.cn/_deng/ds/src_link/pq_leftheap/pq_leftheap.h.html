0009 #include "PQ/PQ.h" //引入优先级队列ADT
0010 #include "BinTree/BinTree.h" //引入二叉树节点模板类
0011 
0012 template <typename T>
0013 class PQ_LeftHeap : public PQ<T>, public BinTree<T> { //基于二叉树，以左式堆形式实现的PQ
0014 public:
0015    PQ_LeftHeap() { } //默认构造
0016    PQ_LeftHeap ( T* E, int n ) //批量构造：可改进为Floyd建堆算法
0017    {  for ( int i = 0; i < n; i++ ) insert ( E[i] );  }
0018    void insert ( T ); //按照比较器确定的优先级次序插入元素
0019    T getMax(); //取出优先级最高的元素
0020    T delMax(); //删除优先级最高的元素
0021 }; //PQ_LeftHeap
