0009 static void //通过遍历获取各字符的编码
0010 generateCT ( Bitmap* code, int length, HuffTable* table, BinNodePosi ( HuffChar ) v ) {
0011    if ( IsLeaf ( *v ) ) //若是叶节点（还有多种方法可以判断）
0012       {  table->put ( v->data.ch, code->bits2string ( length ) ); return;  }
0013    if ( HasLChild ( *v ) ) //Left = 0
0014       { code->clear ( length ); generateCT ( code, length + 1, table, v->lc ); }
0015    if ( HasRChild ( *v ) ) //Right = 1
0016       { code->set ( length ); generateCT ( code, length + 1, table, v->rc ); }
0017 }
0018 
0019 HuffTable* generateTable ( HuffTree* tree ) { //将各字符编码统一存入以散列表实现的编码表中
0020    HuffTable* table = new HuffTable; Bitmap* code = new Bitmap;
0021    generateCT ( code, 0, table, tree->root() ); release ( code ); return table;
0022 }; //release()负责释放复杂结构，与算法无直接关系，具体实现详见代码包
