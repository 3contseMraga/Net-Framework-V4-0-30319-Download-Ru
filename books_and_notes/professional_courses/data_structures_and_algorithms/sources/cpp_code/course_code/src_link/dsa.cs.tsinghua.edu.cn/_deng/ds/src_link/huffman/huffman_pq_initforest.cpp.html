0009 /******************************************************************************************
0010  * 初始化Huffman森林：根据传入的频率统计表freq，为每个字符创建一棵树
0011  ******************************************************************************************
0012  * forest基于优先级队列实现，此算法适用于符合PQ接口的任何实现方式
0013  * 为Huffman_PQ_List、Huffman_PQ_ComplHeap和Huffman_PQ_LeftHeap共用
0014  * 编译前对应工程只需设置相应标志：DSA_PQ_List、DSA_PQ_ComplHeap或DSA_PQ_LeftHeap
0015  ******************************************************************************************/
0016 HuffForest* initForest ( int* freq ) {
0017    HuffForest* forest = new HuffForest; //以PQ实现的Huffman森林
0018    for ( int i = 0; i < N_CHAR; i++ ) { //为每个字符
0019       BinTree<HuffChar>* tree = new BinTree<HuffChar>(); //生成一棵树，并将字符集频率
0020       tree->insertAsRoot ( HuffChar ( 0x20 + i, rand() % 123 ) ); //存入其中
0021       forest->insert ( tree ); //再将新树插入森林
0022    }
0023    return forest;
0024 }
