0010  * 二叉树接口

0012 
0013 package dsa;
0014 
0015 public interface BinTree {
0016    //返回树根
0017    public BinTreePosition getRoot();
0018 
0019    //判断是否树空
0020    public boolean isEmpty();
0021 
0022    //返回树的规模（即树根的后代数目）
0023    public int getSize();
0024 
0025    //返回树（根）的高度
0026    public int getHeight();
0027 
0028    //前序遍历
0029    public Iterator elementsPreorder();
0030 
0031    //中序遍历
0032    public Iterator elementsInorder();
0033 
0034    //后序遍历
0035    public Iterator elementsPostorder();
0036 
0037    //层次遍历
0038    public Iterator elementsLevelorder();
0039 }
