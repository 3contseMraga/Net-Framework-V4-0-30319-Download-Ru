0009 char* removeSpace ( char* s ) { //剔除s[]中的白空格
0010    char* p = s, *q = s;
0011    while ( true ) {
0012       while ( isspace ( *q ) ) q++;
0013       if ( '\0' == *q ) { *p = '\0'; return s; }
0014       *p++ = *q++;
0015    }
0016 }
