0009 #include<iostream>
0010 using namespace std;
0011 
0012 __int64 powerBF ( __int64, int );
0013 __int64 power ( __int64, int );
0014 __int64 power2BF_I ( int );
0015 __int64 power2_I ( int );
0016 __int64 power2BF ( int );
0017 __int64 power2 ( int );
0018 
0019 /******************************************************************************************
0020  * 测试Power
0021  ******************************************************************************************/
0022 int main ( int argc, char* argv[] ) {
0023    __int64 a; int r;
0024    if ( 3 > argc ) {  cout << "Usage: " << argv[0] << "<base><exponent>" << endl; return 1;  } //检查参数
0025    a = _atoi64 ( argv[1] ); r = atoi ( argv[2] ); //获取输入
0026    printf ( "\n2 ^ %d  =  %I64d  =  %I64d\n\n", r, power2BF_I ( r ), power2_I ( r ) ); //迭代版
0027    printf ( "\n2 ^ %d  =  %I64d  =  %I64d\n\n", r, power2BF ( r ), power2 ( r ) ); //递归版
0028    printf ( "\n%I64d ^ %d  =  %I64d  =  %I64d\n\n", a, r, powerBF ( a, r ), power ( a, r ) ); //通用算法迭代版
0029    return 0;
0030 }
