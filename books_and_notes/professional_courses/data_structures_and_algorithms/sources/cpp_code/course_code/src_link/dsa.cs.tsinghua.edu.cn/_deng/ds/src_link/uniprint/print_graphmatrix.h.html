0012 template <typename Tv, typename Te> //顶点类型、边类型
0013 void UniPrint::p ( GraphMatrix<Tv, Te>& s ) { //引用
0014    int inD = 0; for ( int i = 0; i < s.n; i++ ) inD += s.inDegree ( i );
0015    int outD = 0; for ( int i = 0; i < s.n; i++ ) outD += s.outDegree ( i );
0016    printf ( "%s[%d]*(%d, %d):\n", typeid ( s ).name(), &s, s.n, s.e ); //基本信息
0017 // 标题行
0018    print ( s.n ); printf ( " " ); print ( inD ); printf ( "|" );
0019    for ( int i = 0; i < s.n; i++ ) { print ( s.vertex ( i ) ); printf ( "[" ); print ( s.status ( i ) ); printf ( "] " ); }
0020    printf ( "\n" );
0021 // 标题行（续）
0022    print ( outD ); printf ( " " ); print ( s.e ); printf ( "|" );
0023    for ( int i = 0; i < s.n; i++ ) { print ( s.inDegree ( i ) ); printf ( " " ); }
0024    printf ( "| dTime fTime Parent Weight\n" );
0025 // 水平分隔线
0026    printf ( "-----------+" ); for ( int i = 0; i < s.n; i++ ) printf ( "------" );
0027    printf ( "+----------------------------\n" );
0028 // 逐行输出各顶点
0029    for ( int i = 0; i < s.n; i++ ) {
0030       print ( s.vertex ( i ) ); printf ( "[" ); print ( s.status ( i ) ); printf ( "] " ); print ( s.outDegree ( i ) ); printf ( "|" );
0031       for ( int j = 0; j < s.n; j++ )
0032          if ( s.exists ( i, j ) ) { print ( s.edge ( i, j ) ); print ( s.type ( i, j ) ); }
0033          else printf ( "     ." );
0034       printf ( "| " ); print ( s.dTime ( i ) ); printf ( " " ); print ( s.fTime ( i ) );
0035       printf ( "     " ); if ( 0 > s.parent ( i ) ) print ( "^" ); else print ( s.vertex ( s.parent ( i ) ) );
0036       printf ( "  " ); if ( INT_MAX > s.priority ( i ) ) print ( s.priority ( i ) ); else print ( " INF" );
0037       printf ( "\n" );
0038    }
0039    printf ( "\n" );
0040 }
