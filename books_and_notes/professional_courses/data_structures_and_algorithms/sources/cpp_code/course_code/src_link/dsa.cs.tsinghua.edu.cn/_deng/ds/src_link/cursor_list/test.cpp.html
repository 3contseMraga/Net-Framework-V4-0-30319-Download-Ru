0014 /******************************************************************************************
0015  * 测试游标式列表
0016  ******************************************************************************************/
0017 void testCursor ( int testSize ) {
0018    CursorList<char> CL ( testSize ); CL.print();
0019    //CL.insert('A'); CL.insert('B'); CL.insert('C'); CL.insert('D'); CL.insert('E'); CL.print();
0020    //CL.remove(CL.find('A')); CL.remove(CL.find('B')); CL.remove(CL.find('C')); CL.remove(CL.find('D')); CL.remove(CL.find('E')); CL.print();
0021    //CL.insert('A'); CL.insert('B'); CL.insert('C'); CL.insert('D'); CL.insert('E'); CL.print();
0022    for ( int i = 0; i < testSize ; i++ ) {
0023       char x = 'A' + ( rand() % 26 );
0024       if ( 30 < ( rand() % 100 ) ) {
0025          printf ( "%c inserted at %d\n", x, CL.insert ( x ) );
0026       } else {
0027          if ( 0 > CL.find ( x ) )
0028             printf ( "%c not found\n", x );
0029          else
0030             printf ( "%c removed at %d\n", x, CL.remove ( CL.find ( x ) ) );
0031       }
0032       CL.print();
0033    }
0034    for ( char x = 'A'; x <= 'Z'; x++ ) {
0035       while ( 0 <= CL.find ( x ) )
0036          printf ( "%c removed at %d\n", x, CL.remove ( CL.find ( x ) ) );
0037       CL.print();
0038    }
0039 }
