0009 #include "Vector/Vector.h" //借助多重继承机制，基于向量
0010 #include "PQ/PQ.h" //按照优先级队列ADT实现的
0011 template <typename T> struct PQ_ComplHeap : public PQ<T>, public Vector<T> { //完全二叉堆
0012    PQ_ComplHeap() { } //默认构造
0013    PQ_ComplHeap ( T* A, Rank n ) { copyFrom ( A, 0, n ); heapify ( _elem, n ); } //批量构造
0014    void insert ( T ); //按照比较器确定的优先级次序，插入词条
0015    T getMax(); //读取优先级最高的词条
0016    T delMax(); //删除优先级最高的词条
0017 }; //PQ_ComplHeap
