0009 template <typename T> int List<T>::deduplicate() {
0010    int oldSize = _size; ListNodePosi(T) p = first(); ListNodePosi(T) q = NULL;
0011    for ( Rank r = 0; p != trailer; p = p->succ, q = find ( p->data, r, p ) )
0012       q ? remove ( q ): r++; //r为无重前缀的长度
0013    return oldSize - _size; //即被删除元素总数
0014 }
