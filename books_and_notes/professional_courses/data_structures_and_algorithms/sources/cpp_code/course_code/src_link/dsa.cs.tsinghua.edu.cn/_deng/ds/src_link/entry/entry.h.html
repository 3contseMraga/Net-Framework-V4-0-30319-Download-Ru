0009 template <typename K, typename V> struct Entry { //词条模板类
0010    K key; V value; //关键码、数值
0011    Entry ( K k = K(), V v = V() ) : key ( k ), value ( v ) {}; //默认构造函数
0012    Entry ( Entry<K, V> const& e ) : key ( e.key ), value ( e.value ) {}; //基于克隆的构造函数
0013    bool operator< ( Entry<K, V> const& e ) { return key <  e.key; }  //比较器：小于
0014    bool operator> ( Entry<K, V> const& e ) { return key >  e.key; }  //比较器：大于
0015    bool operator== ( Entry<K, V> const& e ) { return key == e.key; } //判等器：等于
0016    bool operator!= ( Entry<K, V> const& e ) { return key != e.key; } //判等器：不等于
0017 }; //得益于比较器和判等器，从此往后，不必严格区分词条及其对应的关键码
