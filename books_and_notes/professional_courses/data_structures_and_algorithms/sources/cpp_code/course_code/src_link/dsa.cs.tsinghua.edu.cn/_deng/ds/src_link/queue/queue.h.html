0009 #include "List/List.h" //以List为基类
0010 template <typename T> class Queue: public List<T> { //队列模板类（继承List原有接口）
0011 public: //size()、empty()以及其它开放接口均可直接沿用
0012    void enqueue ( T const& e ) { insertAsLast ( e ); } //入队：尾部插入
0013    T dequeue() { return remove ( first() ); } //出队：首部删除
0014    T& front() { return first()->data; } //队首
0015 };
