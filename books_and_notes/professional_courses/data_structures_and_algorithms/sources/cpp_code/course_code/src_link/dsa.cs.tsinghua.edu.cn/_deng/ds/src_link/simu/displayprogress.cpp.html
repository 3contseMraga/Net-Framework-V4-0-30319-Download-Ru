0011 /******************************************************************************************
0012  * 显示当前各（窗口）队列情况
0013  ******************************************************************************************/
0014 void displayProgress ( Queue<Customer> windows[], int nWin, int now ) {
0015    system ( "cls" ); //清屏
0016    printf ( "============ Time: %4d ============\n", now );
0017    for ( int i = 0; i < nWin; i++ ) { //对每个窗口，分别
0018       printf ( "Window #%c:  ", 'A' + i ); //输出窗口编号
0019       windows[i].traverse ( printCustomer ); printf ( "\n" ); //列出队列中的所有顾客
0020    }
0021 }
