0009 /*
0010  * 映射结构接口
0011  */
0012 
0013 package  dsa;
0014 
0015 public interface Map {
0016    //查询映射结构当前的规模
0017    public int getSize();
0018 
0019    //判断映射结构是否为空
0020    public boolean isEmpty();
0021 
0022    //若映射中存在以key为关键码的条目，则返回该条目的数据对象；否则，返回null
0023    public Object get(Object key);
0024 
0025    //若映射中不存在以key为关键码的条目，则插入条目(key, value)并返回null
0026    //否则，将已有条目的数据对象替换为value，并返回原先的数据对象
0027    public Object put(Object key, Object value);
0028 
0029    //若映射中存在以key为关键码的条目，则删除之并返回其数据对象；否则，返回null
0030    public Object remove(Object key);
0031 
0032    //返回映射中所有条目的一个迭代器
0033    public Iterator entries();
0034 }
