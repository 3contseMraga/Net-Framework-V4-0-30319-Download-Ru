0013 package dsa;
0014 
0015 public interface SortedDictionary extends Dictionary {
0016    //若词典非空，则返回其中关键码最小的条目；否则，返回null
0017    public Entry first();
0018 
0019    //若词典非空，则返回其中关键码最大的条目；否则，返回null
0020    public Entry last();
0021 
0022    //返回由关键码不小于key的条目依非降序组成的迭代器
0023    public Iterator successors(Object key);
0024 
0025    //返回由关键码不大于key的条目依非升序组成的迭代器
0026    public Iterator predecessors(Object key);
0027 }
