0009 template <typename T> //无序向量的顺序查找：返回最后一个元素e的位置；失败时，返回lo - 1
0010 Rank Vector<T>::find ( T const& e, Rank lo, Rank hi ) const { //assert: 0 <= lo < hi <= _size
0011    while ( ( lo < hi-- ) && ( e != _elem[hi] ) ); //从后向前，顺序查找
0012    return hi; //若hi < lo，则意味着失败；否则hi即命中元素的秩
0013 }
