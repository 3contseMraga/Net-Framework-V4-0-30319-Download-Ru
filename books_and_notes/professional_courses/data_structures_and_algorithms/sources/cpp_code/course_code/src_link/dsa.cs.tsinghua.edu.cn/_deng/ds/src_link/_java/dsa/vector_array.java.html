0010  * 基于数组的向量实现
0011  */
0012 
0013 package dsa;
0014 
0015 public class Vector_Array implements Vector {
0016    private final int N = 1024;//数组的容量
0017    private int n = 0;//向量的实际规模
0018    private Object[] A;//对象数组
0019 
0020    //构造函数
0021    public Vector_Array() {
0022       A = new Object[N];
0023       n = 0;
0024    }
0025 
0026    //返回向量中元素数目
0027    public int getSize() { return n; }
0028 
0029    //判断向量是否为空
0030    public boolean isEmpty() { return (0 == n) ? true : false; }
0031 
0032    //取秩为r的元素
0033    public Object getAtRank(int r)//O(1)
0034    throws ExceptionBoundaryViolation {
0035       if (0 > r || r >= n) throw new ExceptionBoundaryViolation("意外：秩越界");
0036       return A[r];
0037    }
0038 
0039    //将秩为r的元素替换为obj
0040    public Object replaceAtRank(int r, Object obj)
0041    throws ExceptionBoundaryViolation {
0042       if (0 > r || r >= n) throw new ExceptionBoundaryViolation("意外：秩越界");
0043       Object bak = A[r];
0044       A[r] = obj;
0045       return bak;
0046    }
0047 
0048    //插入obj，作为秩为r的元素；返回该元素
0049    public Object insertAtRank(int r, Object obj)
0050    throws ExceptionBoundaryViolation {
0051       if (0 > r || r > n)  throw new ExceptionBoundaryViolation("意外：秩越界");
0052       if (n >= N) throw new ExceptionBoundaryViolation("意外：数组溢出");
0053       for (int i = n; i > r; i--) A[i] = A[i-1]; //后续元素顺次后移
0054       A[r] = obj;//插入
0055       n++;//更新当前规模
0056       return obj;
0057    }
0058 
0059    //删除秩为r的元素
0060    public Object removeAtRank(int r)
0061    throws ExceptionBoundaryViolation {
0062       if (0 > r || r >= n) throw new ExceptionBoundaryViolation("意外：秩越界");
0063       Object bak = A[r];
0064       for (int i = r; i < n; i++) A[i] = A[i+1]; //后续元素顺次前移
0065       n--;//更新当前规模
0066       return bak;
0067    }
0068 }
