0009 typedef int Rank; //秩
0010 #define ListNodePosi(T) ListNode<T>* //列表节点位置
0011 
0012 template <typename T> struct ListNode { //列表节点模板类（以双向链表形式实现）
0013 // 成员
0014    T data; ListNodePosi(T) pred; ListNodePosi(T) succ; //数值、前驱、后继
0015 // 构造函数
0016    ListNode() {} //针对header和trailer的构造
0017    ListNode ( T e, ListNodePosi(T) p = NULL, ListNodePosi(T) s = NULL )
0018       : data ( e ), pred ( p ), succ ( s ) {} //默认构造器
0019 // 操作接口
0020    ListNodePosi(T) insertAsPred ( T const& e ); //紧靠当前节点之前插入新节点
0021    ListNodePosi(T) insertAsSucc ( T const& e ); //紧随当前节点之后插入新节点
0022 };
