0009 template <typename T> Vector<T>& Vector<T>::operator= ( Vector<T> const& V ) { //深复制
0010    if ( _elem ) delete [] _elem; //释放原有内容
0011    copyFrom ( V._elem, 0, V.size() ); //整体复制
0012    return *this; //返回当前对象的引用，以便链式赋值
0013 }
