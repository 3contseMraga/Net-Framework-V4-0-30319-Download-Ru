0012 import dsa.*;
0013 import java.io.*;
0014 
0015 public class TestVector {
0016    static final int TEST_SIZE = 31;
0017    public static void main(String[] args) {
0018       Vector vec = new Vector_ExtArray();
0019       //    Vector vec = new Vector_Array();
0020       while (TEST_SIZE > vec.getSize()) {
0021          double dice = Math.random();
0022          if (0.35 < dice)
0023             vec.insertAtRank((int)(Math.random()*(1 + vec.getSize())), new Integer((int)(Math.random()*(4 * TEST_SIZE))));
0024          else if (!vec.isEmpty())
0025             vec.removeAtRank((int)(Math.random()*vec.getSize()));
0026          Print(vec);
0027       }
0028    }
0029 
0030    //输出向量中的所有元素
0031    protected static void Print(Vector v) {
0032       for (int i = 0; i < v.getSize(); i++)
0033          System.out.print(v.getAtRank(i) + " ");
0034       System.out.println();
0035    }
0036 
0037 }
