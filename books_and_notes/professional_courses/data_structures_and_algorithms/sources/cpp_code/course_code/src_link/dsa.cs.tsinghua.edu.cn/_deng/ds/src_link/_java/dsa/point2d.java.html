0013 package dsa;
0014 
0015 public class Point2D {
0016    private int x, y;
0017 
0018    public Point2D(int xx, int yy)
0019    {  x = xx;  y = yy; }
0020 
0021    public int getX()
0022    {  return x; }
0023 
0024    public int getY()
0025    {  return y; }
0026 }
